custom_button = {
	name = truce_icon
	potential = {  }
	trigger = {  }
	effect = {
		# To display tt for neutral countries
		every_country = {
			limit = { has_ruler_flag = truce_flag_with_@ROOT }
			custom_tooltip = "cannot_declare_war_on_us_tt"
		}
		hidden_effect = {
			if = {
				# Cannot target self
				limit = { NOT = { tag = FROM } }
				if = {
					# Remove neutrality
					limit = {
						has_ruler_flag = truce_flag_with_@FROM
						FROM = { has_ruler_flag = truce_flag_with_@ROOT }
					}
					clr_ruler_flag = truce_flag_with_@FROM
					FROM = { clr_ruler_flag = truce_flag_with_@ROOT }
				}
				# Add neutrality
				else = {
					set_ruler_flag = truce_flag_with_@FROM
					FROM = { set_ruler_flag = truce_flag_with_@ROOT }
				}
			}
		}
	}
	tooltip = "neutrality_button_tt"
}
