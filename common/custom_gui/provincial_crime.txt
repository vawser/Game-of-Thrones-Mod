custom_button = {
	name = criminal_rating
	potential = { NOT = { has_any_crime_lvl = yes } }
	trigger = { hidden_trigger = { owned_by = FROM } }
	effect = {  }
	tooltip = "criminal_rating_tt"
}

custom_button = {
	name = criminal_rating_active
	potential = { has_any_crime_lvl = yes }
	trigger = { hidden_trigger = { owned_by = FROM } }
	effect = {
		if = {
			limit = {
				owned_by = FROM
				NOT = { has_province_modifier = anticrime_campaign }
				has_any_crime_lvl = yes
			}
			hidden_effect = {
				export_to_variable = {
					which = counter
					value = development
				}
			}
			if = {
				limit = { units_in_province = owner }
				hidden_effect = {
					multiply_variable = {
						which = counter
						value = 0.5
					}
				}
				custom_tooltip = "discounted_by_troops_tt"
			}
			decrease_crime_lvl_medium = yes
			add_province_modifier = {
				name = "anticrime_campaign"
				duration = 1825
			}
			hidden_effect = {
				while = {
					limit = { check_variable = { which = counter value = 0 } }
					FROM = { add_treasury = -10 }
					change_variable = {
						which = counter
						value = -1
					}
				}
			}
		}
	}
	tooltip = "criminal_rating_tt"
}
