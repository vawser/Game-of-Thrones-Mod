king_land_the_royal_court = {
	empire = hre
	required_reform = king_introduce_the_new_crown_tax
	potential = {
		has_dlc = "Emperor"
	}
	emperor = {
		diplomats = 1
		free_city_imperial_authority = 0.5
	}
	on_effect = {
		custom_tooltip = "ROYAL_COURT_ESTABLISHED"
		hidden_effect = {
			if = {
				limit = {
					226 = {
						is_part_of_hre = yes
						is_capital = yes
					}
				}
				226 = {
					province_event = {
						id = sk_event.1
					}
					add_province_triggered_modifier = perpetual_diet
				}
			}
			else = {
				random_province = {
					limit = {
						is_part_of_hre = yes
						is_capital = yes
						OR = {
							AND = {
								has_dlc = "Art of War"
								owner = { has_reform = free_city }
							}
							AND = {
								owner = { is_elector = yes }
								development = 20
							}
							development = 30
						}
					}
					province_event = {
						id = sk_event.1
					}
					add_province_triggered_modifier = perpetual_diet
				}
			}
		}
	}
	off_effect = {
		every_province = {
			remove_province_triggered_modifier = perpetual_diet
		}
	}
	gui_container = mainline
}
