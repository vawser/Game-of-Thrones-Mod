request_royal_submission = {
	category = influence
	
	alert_index = 40
	alert_tooltip = request_royal_submission_tooltip
	
	require_acceptance = yes # Whether the recipient gets an option to decline
	
	is_visible = {
		is_subject = no
        has_reform = noble_house_reform
        
		FROM = {
			has_reform = royal_house_reform
		}
	}
	is_allowed = {
		is_subject = no
        has_reform = noble_house_reform
        
		FROM = {
			has_reform = royal_house_reform
		}
}
	on_accept = {
		FROM = {
			create_subject = {
				subject_type = royal_subject
				subject = ROOT
			}
		}
	}
	on_decline = {
        
	}
	
	ai_acceptance = {
		add_entry = {
			name = AIWEIGHT_BASE
			limit = {
				always = yes
			}
			export_to_variable = {
				variable_name = ai_value
				value = 10
			}
		}
		add_entry = {
			name = AIWEIGHT_PRESTIGE
			limit = {
				always = yes
			}
			export_to_variable = {
				variable_name = ai_value
				value = prestige
			}
			subtract_variable = { which = ai_value value = 0 } # Means 0 prestige is the zero point
		}
		add_entry = {
			name = AIWEIGHT_OPINION
			limit = {
				always = yes
			}
			export_to_variable = {
				variable_name = ai_value
				value = opinion
				who = FROM
				with = ROOT
			}
			subtract_variable = { which = ai_value value = 0 } # Means 0 opinion is the zero point
		}
	}

	ai_will_do = {
		always = yes
	}
}
