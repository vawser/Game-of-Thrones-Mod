bribe_for_support = {
	category = relation

	alert_index = 39
	alert_tooltip = bribe_for_support_alert_tooltip

	require_acceptance = no

	is_visible = {
		OR = { # Only used by the two nations in question
			has_country_flag = petition_for_great_house_subject
			has_country_flag = petition_for_great_house_overlord
		}
		if = { # Only when specific incident is active
			limit = { active_imperial_incident = any }
			active_imperial_incident = incident_change_great_house
		}
		else = { always = no }
		# Can't use on overlord obviously
		FROM = { NOT = { overlord_of = ROOT } ai = yes }
	}
	is_allowed = {
		NOT = {
			reverse_has_opinion_modifier = {
				who = FROM
				modifier = bribed_for_support
			}
		}
		monthly_income = FROM
	}
	on_accept = {
		set_income_variable_with_tag = {
			income_var = briber_income
			percent_amount = 1.0
			tag = FROM
		}
		transfer_treasury = {
			giver = ROOT
			taker = FROM
			variable = briber_income
		}
		FROM = {
			add_opinion = {
				who = ROOT
				modifier = bribed_for_support
				months = 12
			}
			every_subject_country = {
				limit = { is_subject_of_type = feudatory }
				add_opinion = {
					who = ROOT
					modifier = bribed_for_support
					months = 12
				}
			}
		}
	}

	ai_will_do = {
		always = no
	}
}

establish_neutrality_pact = {
	category = relation

	alert_index = 39
	alert_tooltip = bribe_for_support_alert_tooltip

	require_acceptance = yes
	
	is_visible = {
		ai = no
	}
	is_allowed = {
		custom_trigger_tooltip = {
			tooltip = "neutality_pact_active_tt"
			NOT = { has_neutrality_pact = yes }
		}
	}
	on_accept = {
		add_neutrality_pact = yes
	}

	ai_acceptance = {
		add_entry = {
			name = AIWEIGHT_BASE
			export_to_variable = {
				variable_name = ai_value
				value = -20
			}
		}
		add_entry = {
			name = NEIGHBOR_COUNTRY_STARTS
			limit = {
				is_neighbor_of = FROM
			}
			export_to_variable = {
				variable_name = ai_value
				value = 30
			}
		}
		add_entry = {
			name = diplomatic_reputation
			limit = { }
			export_to_variable = {
				variable_name = ai_value
				value = trigger_value:diplomatic_reputation
			}
			#1 rep = 9
			multiply_variable = {
				which = ai_value
				value = 9
			}
		}
		add_entry = {
			name = TRUST
			export_to_variable = {
				variable_name = ai_value
				value = trust
				who = FROM
				with = ROOT
			}			
			subtract_variable = { 
				which = ai_value
				value = 50
			}
			#1 trust = +4 ai points
			multiply_variable = {
				which = ai_value
				value = 0.4
			}
		}
		add_entry = {
			name = OPINION
			export_to_variable = {
				variable_name = ai_value
				value = opinion
				who = FROM
				with = ROOT
			}
			multiply_variable = {
				which = ai_value 
				value = 0.2
			}
		}
	}

	ai_will_do = {
		always = no
	}
}
