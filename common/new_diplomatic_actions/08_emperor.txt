remove_from_the_council = {
	category = emperor

	alert_index = 40
	alert_tooltip = remove_from_the_council_alert_tooltip

	require_acceptance = no

	is_visible = {
		OR = {
			is_emperor = yes
			has_council_position = { sm_council_position = handoftheking }
		}
		event_target:IronThroneKing = { has_estate_privilege = estate_nobles_the_small_council }
		FROM = {
			is_part_of_hre = yes
			is_on_the_small_council = yes
			is_emperor = no
			NOT = { tag = FAT }
		}
	}
	is_allowed = {
		always = yes
	}
	on_accept = {
		FROM = {
			clear_council_country = yes
		}
	}

	ai_will_do = {
		always = no
	}
}

appoint_hand_of_the_king = {
	category = emperor

	alert_index = 40
	alert_tooltip = appoint_hand_of_the_king_alert_tooltip

	require_acceptance = yes

	is_visible = {
		is_emperor = yes
		event_target:IronThroneKing = { has_estate_privilege = estate_nobles_the_small_council }
		FROM = {
			is_part_of_hre = yes
			NOT = { has_council_position = { sm_council_position = handoftheking } }
			is_emperor = no
			NOT = { tag = FAT }
		}
	}
	is_allowed = {
		always = yes
	}
	on_accept = {
		FROM = { set_council_position = { sm_council_position = handoftheking } }
	}
	on_decline = {

	}

	ai_acceptance = {
		add_entry = {
			name = TRUST
			export_to_variable = {
				variable_name = ai_value
				value = trust
				who = FROM
				with = ROOT
			}			
			subtract_variable = { 
				which = ai_value
				value = 50
			}
			#1 trust = +4 ai points
			multiply_variable = {
				which = ai_value
				value = 0.4
			}
		}
		add_entry = {
			name = OPINION
			export_to_variable = {
				variable_name = ai_value
				value = opinion
				who = FROM
				with = ROOT
			}
			multiply_variable = {
				which = ai_value 
				value = 0.2
			}
		}
		add_entry = {
			name = diplomatic_reputation
			limit = { }
			export_to_variable = {
				variable_name = ai_value
				value = trigger_value:diplomatic_reputation
			}
			#1 rep = 3
			multiply_variable = {
				which = ai_value
				value = 3
			}
		}
		add_entry = {
			name = AIWEIGHT_POWERF
			export_to_variable = {
				variable_name = ai_value
				value = 1
			}
			multiply_variable = {
				which = ai_value
				value = 50
			}
		}
		add_entry = {
			name = AIWEIGHT_EXPECTS
			limit = {
				FROM = { has_reform = great_house_reform }
			}
			export_to_variable = {
				variable_name = ai_value
				value = 1
			}
			multiply_variable = {
				which = ai_value
				value = 1000
			}
		}
	}
	ai_will_do = {
		always = yes
	}
}
