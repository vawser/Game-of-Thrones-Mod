# Usurp Overlord
po_usurp_overlord = {
	category = 6
	power_projection = humiliated_rival
	power_cost_base = 1.0
	prestige_base = 0.5
	ae_base = 0.3
	requires_is_allowed = yes
	
	warscore_cost = {
		all_provinces = 0.0
		no_provinces = 80.0
		owner_keeps = 0.0
		conquered = 0.0
		returned_core = 0.0
		released_nation = 0.0
		cancelled_subject = 0.0
		concede_colonial_region = 0.0
	}
	
	is_visible = {
		always = yes
	}
	is_allowed = {
		always = yes
	}
	effect = {
		add_stability = 2
		FROM = {
			elector = no # Remove old Great House status
			add_government_reform = noble_house_reform
			every_subject_country = { # Transfer all Feudatories
				limit = { is_subject_of_type = feudatory }
				ROOT = {
					create_subject = {
						subject_type = feudatory
						who = PREV
					}
				}
			}
			rename_country_usurp_overlord_effect = yes
		}
		elector = yes # Assign new Great House status
		add_government_reform = great_house_reform
		create_subject = {
			subject_type = feudatory
			who = FROM
		}
		# Cleanup Code
		remove_temp_overlord_flag_effect = yes # Clear region country flags
		every_subject_country = { # Convert existing vassals to feudatories
			limit = {
				is_subject_of_type = vassal
				OR = {
					is_part_of_hre = yes
					is_dynasty_government = yes
				}
			}
			ROOT = {
				create_subject = {
					subject_type = feudatory
					who = PREV
				}
			}
		}
	}
	ai_weight = {
		export_to_variable = {
			variable_name = ai_value
			value = 100
		}
	}
}
