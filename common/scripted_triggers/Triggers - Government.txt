westeros_government_king = {
	if = {
		limit = { hre_size = 1 }
		is_emperor = yes
	}
	else = {
		OR = {
			has_reform = royal_house_reform
			has_reform = kingdom_of_westeros_reform
		}
	}
}

westeros_government_great_house = {
	if = {
		limit = { hre_size = 1 }
		is_elector = yes
	}
	else = {
		has_reform = great_house_reform
	}
}
