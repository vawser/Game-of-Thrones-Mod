# General Hospital
country_event = {
	id = devotion_events.1
	title = devotion_events.1.t
	desc = devotion_events.1.d
	picture = WOUNDED_SOLDIERS_eventPicture
	
	trigger = {
		has_dlc = "Common Sense"
		uses_devotion = yes
		NOT = {
			has_country_modifier = dev_general_hospital
		}
	}
	
	

	is_triggered_only = yes
	
	mean_time_to_happen = {
		days = 1
	}
		
	option = {
		name = devotion_events.1.a
		ai_chance = {
			factor = 55
			modifier = {
				factor = 0
				has_dlc = "The Cossacks"
				has_estate = estate_clergy
				NOT = {
					estate_loyalty = {
						estate = estate_clergy
						loyalty = 40
					}
				}
			}
		}
		if = {
			limit = {
				has_dlc = "The Cossacks"
				has_estate = estate_clergy
			}
			add_estate_loyalty = {
				estate = estate_clergy
				loyalty = -10
			}
		}
		add_country_modifier = {
			name = dev_general_hospital
			duration = 3650
		}
	}
	
	option = {
		name = devotion_events.1.b
		ai_chance = {
			factor = 45
		}
		if = {
			limit = {
				has_dlc = "The Cossacks"
				has_estate = estate_clergy
			}
			add_estate_loyalty = {
				estate = estate_clergy
				loyalty = 10
			}
		}
		add_devotion = 10
	}
}
