# Half the Loyalty, Double the Agent
country_event = {
	id = GOT_ESPIONAGE.1
	title = "GOT_ESPIONAGE.1.title"
	desc = "GOT_ESPIONAGE.1.desc"
	picture = SPY_eventPicture
	goto = espionage_event_country_capital

	trigger = {
		has_dlc = "Mare Nostrum"
		any_known_country = {
			has_spy_network_in = {
				who = ROOT
				value = 25
			}
			has_spy_network_from = {
				who = ROOT
				value = 25
			}
			corruption = 10
		}
	}

	is_triggered_only = yes
	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		hidden_effect = {
			capital_scope = {
				save_event_target_as = espionage_event_country_2_capital
			}
			random_known_country = {
				limit = {
					has_spy_network_in = {
						who = ROOT
						value = 25
					}
					has_spy_network_from = {
						who = ROOT
						value = 25
					}
					corruption = 10
				}
				save_event_target_as = espionage_event_country
				capital_scope = {
					save_event_target_as = espionage_event_country_capital
				}
			}
		}
	}

	# Recruit the spy
	option = {
		name = "GOT_ESPIONAGE.1.option.a"
		add_treasury = -15
		event_target:espionage_event_country = {
			add_spy_network_from = {
				who = ROOT
				value = 30
			}
			random = {
				chance = 30
				country_event = {
					id = GOT_ESPIONAGE.2 # Enemy discovers your plot
					days = 120
				}
			}
		}
	}

	# Execute the spy
	option = {
		name = "GOT_ESPIONAGE.1.option.b"
		event_target:espionage_event_country = {
			add_spy_network_in = {
				who = ROOT
				value = -25
			}
		}
	}
}

# $SPYMASTER$ Suspects Neighbors
country_event = {
	id = GOT_ESPIONAGE.8
	title = "GOT_ESPIONAGE.8.title"
	desc = "GOT_ESPIONAGE.8.desc"
	picture = SPY_eventPicture

	trigger = {
		has_dlc = "Mare Nostrum"
		advisor = spymaster
		any_known_country = {
			is_neighbor_of = ROOT
			has_spy_network_in = {
				who = ROOT
				value = 25
			}
		}
		any_known_country = {
			is_neighbor_of = ROOT
			NOT = {
				has_spy_network_in = {
					who = ROOT
					value = 1
				}
			}
		}
	}

	is_triggered_only = yes
	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		hidden_effect = {
			random_list = {
				50 = {
					random_known_country = {
						limit = {
							is_neighbor_of = ROOT
							has_spy_network_in = {
								who = ROOT
								value = 25
							}
						}
						save_event_target_as = neighbor_1
						capital_scope = {
							save_event_target_as = neighbor_1_capital
						}
					}
					random_known_country = {
						limit = {
							is_neighbor_of = ROOT
							NOT = {
								has_spy_network_in = {
									who = ROOT
									value = 1
								}
							}
						}
						save_event_target_as = neighbor_2
						capital_scope = {
							save_event_target_as = neighbor_2_capital
						}
					}
				}
				50 = {
					random_known_country = {
						limit = {
							is_neighbor_of = ROOT
							has_spy_network_in = {
								who = ROOT
								value = 25
							}
						}
						save_event_target_as = neighbor_2
						capital_scope = {
							save_event_target_as = neighbor_2_capital
						}
					}
					random_known_country = {
						limit = {
							is_neighbor_of = ROOT
							NOT = {
								has_spy_network_in = {
									who = ROOT
									value = 1
								}
							}
						}
						save_event_target_as = neighbor_1
						capital_scope = {
							save_event_target_as = neighbor_1_capital
						}
					}
				}
			}
		}
	}

	# 
	option = {
		name = "GOT_ESPIONAGE.8.option.a"
		goto = neighbor_1_capital
		event_target:neighbor_1 = {
			add_opinion = {
				who = ROOT
				modifier = counterintelligence_operation
			}
			hidden_effect = {
				if = {
					limit = {
						has_spy_network_in = {
							who = ROOT
							value = 25
						}
					}
					add_spy_network_in = {
						who = ROOT
						value = -25
					}
				}
			}
		}
	}

	# 
	option = {
		name = "GOT_ESPIONAGE.8.option.b"
		goto = neighbor_2_capital
		event_target:neighbor_2 = {
			add_opinion = {
				who = ROOT
				modifier = counterintelligence_operation
			}
			hidden_effect = {

					limit = {
						has_spy_network_in = {
							who = ROOT
							value = 25
						}
					}
					add_spy_network_in = {
						who = ROOT
						value = -25
					}
				}
