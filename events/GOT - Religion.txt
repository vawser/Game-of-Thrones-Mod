# Restore the Great Sept of Baelor
country_event = {
	id = GOT_RELIGION.9
	title = GOT_RELIGION.9.title
	desc = GOT_RELIGION.9.desc
	picture = RELIGION_eventPicture
	
	trigger = {
        always = no
		NOT = { exists = FAT }
        owns = 27
        NOT = { has_country_flag = offered_restore_the_faith }
	}
	
    mean_time_to_happen = {
		months = 24
	}
    
    immediate = {
        set_country_flag = offered_restore_the_faith
    }
    
	option = {
		name = GOT_RELIGION.9.option.a
		
		27 = {
            cede_province = FAT
            add_core = FAT
        }
		
        custom_tooltip = tt_the_faith_restored_pleased
        hidden_effect = { 
            every_known_country = {
                limit = {
                    capital_scope = {
                        continent = westeros
                    }
                    government = westeros_monarchy
                    religion = asoiaf_faith_of_the_seven
                }
                add_opinion = {
                    who = ROOT
                    modifier = restored_the_faith
                }
            }
        }
        
		ai_chance = {
			factor = 100
		}
	}
	
	option = {
		name = GOT_RELIGION.9.option.b
		
		add_stability = -3
		
        custom_tooltip = tt_the_faith_restored_angry
        hidden_effect = { 
            every_known_country = {
                limit = {
                    capital_scope = {
                        continent = westeros
                    }
                    government = westeros_monarchy
                    religion = asoiaf_faith_of_the_seven
                }
                add_opinion = {
                    who = ROOT
                    modifier = failed_to_restore_the_faith
                }
            }
        }
        
		ai_chance = {
			factor = 0
		}
	}
}
