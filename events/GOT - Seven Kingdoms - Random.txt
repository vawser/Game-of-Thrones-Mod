country_event = {
	id = GOT_SK_RANDOM.1
	title = GOT_SK_RANDOM.1.title
	desc = GOT_SK_RANDOM.1.desc
	picture = ADVISOR_eventPicture
	
	trigger = {
        has_dlc = "Mandate of Heaven"
		any_owned_province = {
			is_capital = no
			is_state = yes
			NOT = { has_province_modifier = sk_local_trade_regulations }
			NOT = { has_province_modifier = sk_local_trade_down_turn }
		}
		is_emperor_of_china = yes
	}
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			random_owned_province = { #Atleast here
				limit = {
					is_capital = no
					is_state = yes
					NOT = { has_province_modifier = sk_local_trade_regulations }
					NOT = { has_province_modifier = sk_local_trade_down_turn }
				}
				save_event_target_as = trade_problems_province
			}
			random_owned_province = { #But rather here
				limit = {
					is_capital = no
					has_building = marketplace
					NOT = { has_province_modifier = sk_local_trade_regulations }
					NOT = { has_province_modifier = sk_local_trade_down_turn }
				}
				save_event_target_as = trade_problems_province
			}
			random_owned_province = { #Here would be even better
				limit = {
					is_capital = no
					has_port = yes
					NOT = { has_province_modifier = sk_local_trade_regulations }
					NOT = { has_province_modifier = sk_local_trade_down_turn }
				}
				save_event_target_as = trade_problems_province
			}
			random_owned_province = { #Warmer
				limit = {
					is_capital = no
					has_port = yes
					has_building = marketplace
					NOT = { has_province_modifier = sk_local_trade_regulations }
					NOT = { has_province_modifier = sk_local_trade_down_turn }
				}
				save_event_target_as = trade_problems_province
			}
			random_owned_province = { #Best
				limit = {
					is_capital = no
					province_has_center_of_trade_of_level = 1
					NOT = { has_province_modifier = sk_local_trade_regulations }
					NOT = { has_province_modifier = sk_local_trade_down_turn }
				}
				save_event_target_as = trade_problems_province
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = GOT_SK_RANDOM.1.option.a
		
		reduce_mandate_effect = yes
        
		event_target:trade_problems_province = {
			add_province_modifier = {
				name = "sk_local_trade_regulations"
				duration = 3650
			}
		}
	}
	
	option = {
		name = GOT_SK_RANDOM.1.option.b
		
		add_mandate_effect = yes
        
		event_target:trade_problems_province = {
			add_province_modifier = {
				name = "sk_local_trade_down_turn"
				duration = 3650
			}
		}
	}
}

		hidden_effect = {

				}
				save_event_target_as = prosperity_boost_here
			}
		}

	}

                limit = { NOT = { prosperity = 100 } }
                add_prosperity = 25
            }
            area = {
                add_province_modifier = {
                    name = "sk_lowered_taxes_in_state"
                    duration = 1825
                }
            }
        }
	}
	
	option = {
