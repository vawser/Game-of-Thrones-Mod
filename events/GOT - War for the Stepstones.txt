# War for the Stepstones
country_event = {
	id = got_war_for_the_stepstones.1
	title = got_war_for_the_stepstones.1.title
	desc = got_war_for_the_stepstones.1.desc
	picture = GOOD_WITH_MONARCH_eventPicture
	
    fire_only_once = yes
    
	trigger = {
        NOT = { has_global_flag = shattered_world }
        tag = A10
        is_year = 106
        NOT = { is_year = 108 }
        exists = X01
        X01 = {
            num_of_cities = 2
        }
    }
	
    mean_time_to_happen = {
		months = 1
    }
    
    immediate = {
        hidden_effect = {
            X01 = {
                random_owned_province = {
                    limit = { is_capital = no }
                    save_event_target_as = assaulted_province
                }
            }
        }
    }
    
	option = {
		name = got_war_for_the_stepstones.1.option.a

        event_target:assaulted_province = {
            cede_province = AB2
            add_core = AB2
        }
        
        release = A43
        
        AB2 = {
            vassalize = A43
        }
        
        hidden_effect = {
            A10 = {
                country_event = { id = got_war_for_the_stepstones.2 days = 1 }
            }
            
            AB2 = {
                country_event = { id = got_war_for_the_stepstones.3 days = 1 }
            }
        }
	}
}
