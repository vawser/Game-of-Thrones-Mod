				limit = {
					mil = 6
				}

						random_list = {
							1 = { change_adm = 1 }
							1 = { change_dip = 1 }
						}
					}
					1 = {}
				}
			}
			if = {
				limit = {
					dip = 6
				}
				random_list = {
					1 = {
						random_list = {
							1 = { change_adm = 1 }
							1 = { change_mil = 1 }
						}
					}
					1 = {}
				}
			}
			if = {
				limit = {
					adm = 6
				}
				random_list = {
					1 = {
						random_list = {
							1 = { change_dip = 1 }
							1 = { change_mil = 1 }
						}
					}
					1 = {}
				}
