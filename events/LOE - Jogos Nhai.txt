country_event = { #Does not work if more than 1 nation has more dev
	id = jogosnhai.4
	title = "jogosnhai.4.t"
    desc = "jogosnhai.4.d"
	picture = BIG_BOOK_eventPicture
	is_triggered_only = yes

	trigger = {
		has_government_attribute = loe_is_jogos_leader
	}

	immediate = {
		clear_global_event_target = loe_jhattar
		clr_ruler_flag = loe_new_prathama
		every_country = {
			limit = {
				has_country_flag = loe_not_enough_dev
			}
			clr_country_flag = loe_not_enough_dev
		}
	}

    option = {
		name = "jogosnhai.4.t1"
		custom_tooltip = "jogosnhai.4.d1"
		hidden_effect = {
			every_country = {
				limit = {
					is_subject_of_type = loe_jogos_nhai_tribe
					OR = {
						NOT = { has_country_flag = loe_not_enough_dev }
						NOT = { has_ruler_flag = loe_new_prathama }					
					}
				}
				if = {
					limit = {
						total_development = ROOT
					}
					set_ruler_flag = loe_new_prathama
				}
				else = {
					set_country_flag = loe_not_enough_dev
				}
			}
			if = {
				limit = {
					any_country = {
						has_ruler_flag = loe_new_prathama
						is_subject_of_type = loe_jogos_nhai_tribe
					}
				}
				set_country_flag = loe_not_enough_dev
				add_government_reform = loe_sthaurin
				random_country = {
					limit = {
						has_ruler_flag = loe_new_prathama
						is_subject_of_type = loe_jogos_nhai_tribe
					}
					save_global_event_target_as = loe_jhattar
					add_government_reform = loe_prathama
					grant_independence = yes
				}
			}
			else = {
				set_ruler_flag = loe_new_prathama
				save_global_event_target_as = loe_jhattar
				add_absolutism = 10
				add_stability = 1
			}
		}
	}

	after = {
		if = {
			limit = {
				ROOT = {
					NOT = { has_ruler_flag = loe_new_prathama }
				}
			}
			every_country = {
				limit = {
					has_country_flag = loe_not_enough_dev
					is_subject = yes
				}
				grant_independence = yes
				country_event = {
					id = jogosnhai.3
				}
			}
			country_event = {
				id = jogosnhai.3
			}
		}
	}
}

country_event = {
	id = jogosnhai.6
	title = "jogosnhai.6.t"
    desc = "jogosnhai.6.d"
	picture = BIG_BOOK_eventPicture
	mean_time_to_happen = {
		days = 1
	}

	trigger = {
		war_score = 50
		has_country_modifier = loe_raid
	}

	immediate = {
		hidden_effect = {
			set_variable = {
				which = loe_total_enemy_dev
				value = 0
			}
			every_war_enemy_country = {
				export_to_variable = {
					which = loe_enemy_dev
					value = total_development
					who = PREV
				}
				change_variable = {
					which = loe_total_enemy_dev
					which = loe_enemy_dev
				}
			}
			divide_variable = {
				which = loe_total_enemy_dev
				value = 10
			}
			set_variable = {
				which = loe_countdown
				which = loe_total_enemy_dev
			}
			while = {
				limit = {
					NOT = {
						check_variable = {
							which = loe_countdown
							value = 1
						}
					}
				}
				subtract_variable = {
					which = loe_countdown
					value = 1
				}
				random_owned_province = {
					random_list = {
						1 = { add_base_manpower = 1 }
						1 = { add_base_production = 1 }
						1 = { add_base_tax = 1 }

				overlord = {

			}
