# Give Warparty Bonuses
country_event = {
	id = tribalgov.2
	hidden = yes

	trigger = {
		has_government_attribute = wildling_warparty_bonus
		NOT = { has_country_modifier = united_warparty_gov_modifier }
		OR = {
			calc_true_if = {
				amount = 3
				all_ally = {
					is_wildling_culture = yes
				}
			}
			calc_true_if = {
				amount = 3
				all_subject_country = {
					is_wildling_culture = yes
				}
			}
			AND = {
				calc_true_if = {
					amount = 2
					all_ally = {
						is_wildling_culture = yes
					}
				}
				calc_true_if = {
					amount = 1
					all_subject_country = {
						is_wildling_culture = yes
					}
				}
			}
			AND = {
				calc_true_if = {
					amount = 1
					all_ally = {
						is_wildling_culture = yes
					}
				}
				calc_true_if = {
					amount = 2
					all_subject_country = {
						is_wildling_culture = yes
					}
				}
			}
		}
	}

	mean_time_to_happen = { days = 7 }

	immediate = {
		hidden_effect = {
			add_country_modifier = {
				name = united_warparty_gov_modifier
				duration = -1
				hidden = yes
			}
		}
	}

	option = { }
}

# Remove Warparty Bonus
country_event = {
	id = tribalgov.3
	hidden = yes

	trigger = {
		has_country_modifier = united_warparty_gov_modifier
		OR = {
			NOT = { has_government_attribute = wildling_warparty_bonus }
			AND = {
				NOT = {
					calc_true_if = {
						amount = 3
						all_ally = {
							is_wildling_culture = yes
						}
					}
				}
				NOT = {
					calc_true_if = {
						amount = 3
						all_subject_country = {
							is_wildling_culture = yes
						}
					}
				}
				NOT = {
					AND = {
						calc_true_if = {
							amount = 2
							all_ally = {
								is_wildling_culture = yes
							}
						}
						calc_true_if = {
							amount = 1
							all_subject_country = {
								is_wildling_culture = yes
							}
						}
					}
				}
				NOT = {
					AND = {
						calc_true_if = {
							amount = 1
							all_ally = {
								is_wildling_culture = yes
							}
						}
						calc_true_if = {
							amount = 2
							all_subject_country = {
								is_wildling_culture = yes
							}
						}
					}
				}
			}
		}
	}

	mean_time_to_happen = { days = 7 }

	immediate = {

	}

	option = { }
}
