namespace = got_colonial

province_event = {
	id =  got_colonial.1
	title = got_colonial.1.title
	desc = got_colonial.1.desc
	picture = COLONIZATION_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		owner = { has_idea_group = exploration_ideas }
		has_empty_adjacent_province = yes
		owner = { NOT = { num_of_colonies = 2 } }
	}

	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = got_colonial.1.option.a
		random_empty_neighbor_province = {
			create_colony = 200
		}
	}
	option = {
		name = got_colonial.1.option.b
		trigger = {
			owner = {
			num_of_colonies = 1
			}
		}
		owner = {
			random_owned_province = {
				limit = {
					is_colony = yes
				}
				add_colonysize = 50
			}
		}
	}
}

		trade_goods = fur

		days = 1

	}

		name = got_colonial.6.option.b

	}
}

			}

			owner = { num_of_colonies = 1 }
		}
		owner = {
			random_owned_province = {
				limit = {
					is_colony = yes
				}
				add_colonysize = 50
			}
		}
	}
}

# Colonial Migration
province_event = {
	id = got_colonial.8
	title = got_colonial.8.title
	desc = got_colonial.8.desc
	picture = COLONIZATION_eventPicture

	is_triggered_only = yes

	trigger = {
		colonysize = 300
		NOT = { colonysize = 500 }
		owner = {
			any_owned_province = {
				is_colony = yes
				colonysize = 500
			}
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = got_colonial.8.option.a
		add_colonysize = -200
		owner = {
			random_owned_province = {
				limit = { is_colony = yes colonysize = 500 }
				add_colonysize = 200
			}
		}	
	}
}
